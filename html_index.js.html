/**
 * html解析
 * @param {String} str 字符串
 * html_decode => decodeHtml
 */
export const decodeHtml = (str) => {
  let s = ''
  if (str) {
    if (str.length === 0) {
      return ''
    }
    s = str.replace(/&amp;/g, '&')
    s = s.replace(/&lt;/g, '<')
    s = s.replace(/&gt;/g, '>')
    s = s.replace(/&nbsp;/g, ' ')
    s = s.replace(/&#39;/g, '\'')
    s = s.replace(/&quot;/g, '\'')
    s = s.replace(/<br>/g, '\n')
  }
  return s
}

/**
 * html转义
 * @param {String} str 字符串
 * html_encode => encodeHtml
 */
export const encodeHtml = (str) => {
  let s = ''
  if (str) {
    if (str.length === 0) {
      return ''
    }
    s = str.replace(/&/g, '&amp;')
    s = s.replace(/</g, '&lt;')
    s = s.replace(/>/g, '&gt;')
    s = s.replace(/'/g, '&quot;')
  }
  return s
}

/**
 * 去除标签
 * @param {String} str 字符串
 * removeHTMLTag => removeTag
 */
export const removeTag = (str) => {
  if (str) {
    str = str.toString().replace(/(\n)/g, '')
    str = str.replace(/(\t)/g, '')
    str = str.replace(/(\r)/g, '')
    str = str.replace(/<\/?[^>]*>/g, '')
    str = str.replace(/\s*/g, '')
    return str
  } else {
    return ''
  }
}
